\section*{Question 1}
\begin{itemize}
    \item[] Prove the following statement using a direct proof.
    \begin{itemize}
        \item[1.] If $m + n$ and $n + p$ are odd integers, where $m$, $n$, and $p$ are integers, then $m + p$ is even.
    \begin{proof}
        \begin{theorem}
            If $m + n$ and $n + p$ are odd integers, where $m$, $n$, $p$, $j$, $k$, $l$ are integers, then $m + p$ is even.
        \end{theorem}
        \begin{definition}
            An integer $n$ is odd if there exists an integer $k$ such that $n = 2k + 1$.
            An integer $n$ is even if there exists an integer $k$ such that $n = 2k$.
        \end{definition}
        \begin{align*}
            m + n &= 2k + 1\\
            n + p &= 2j + 1 \\
            m + p &= (m + n) + (n + p) - n - n \\
            &= (2k + 1) + (2j + 1) - n - n \\
            &= 2k + 2j + 2n \\
            &= 2(k + j + n) \\
            l &= (k + j + n) \\
            m + p &= 2l \\
        \end{align*}
        \item[] Since $m + p = 2l$, where $l$ is an integer, $m + p$ is even.
        \end{proof}
        \end{itemize}   
    

    \end{itemize}
